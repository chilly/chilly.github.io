  "fmt"

        // Fetch иҝ”еӣһ URL зҡ„ body еҶ…е®№пјҢе№¶дё”е°ҶеңЁиҝҷдёӘйЎөйқўдёҠжүҫеҲ°зҡ„ URL ж”ҫеҲ°дёҖдёӘ slice дёӯгҖӮ
  Fetch(url string) (body string, urls []string, err error)
}
var lockx = make(chan int,1)
// еҗҢжӯҘйҖҡдҝЎдҪҝз”Ё
func LockFun(f func()) {
  lockx<-1
  f()
  <-lockx
}
var visited map[string]bool = make(map[string]bool)
// Crawl дҪҝз”Ё fetcher д»ҺжҹҗдёӘ URL ејҖе§ӢйҖ’еҪ’зҡ„зҲ¬еҸ–йЎөйқўпјҢзӣҙеҲ°иҫҫеҲ°жңҖеӨ§ж·ұеәҰгҖӮ
func Crawl(url string, depth int, fetcher Fetcher, banner chan int) {

  if depth <= 0 || visited[url] {
    banner<-1
    return
  }
  body, urls, err := fetcher.Fetch(url)
  LockFun(func(){
    visited[url]=true
  })
  fmt.Printf("found: %s %q\n", url, body)
  if err != nil {
    fmt.Println(err)
    banner<-1
    return
  }
  subBanner := make(chan int, len(urls))
  for _, u := range urls {
     // е№¶иЎҢеҗ§пҪһпҪһ 
      go Crawl(u, depth-1, fetcher, subBanner);
  }
  for i:=0; i < len(urls); i++ {
    // subBannerз”ЁжқҘйҳІжӯўйҖҖеҮә
    <-subBanner
  }
  // bannerз”ЁдәҺи®©зҲ¶иҠӮзӮ№йҖҖеҮә
  banner<-1
  return
}

func main() {
  mainBanner := make(chan int,1)
  Crawl("http://golang.org/", 4, fetcher, mainBanner)
  <-mainBanner
}
